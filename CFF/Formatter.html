def self.date_present?(attribute)

  return !attribute.nil? if attribute.is_a?(Date)

end

def self.try_get_month(value)
  if value.instance_of? Date
    value.month
  else
    begin
      date = Date.parse(value.to_s)
      date.month.to_s
    rescue ArgumentError
      nil
    end
  end
end
